
#include<bits/stdc++.h>

#define D(x)        cout<<#x " = "<<(x)<<endl
#define un(x)       x.erase(unique(x.begin(),x.end()), x.end())
#define sf(n)       scanf("%d", &n)
#define sff(a,b)    scanf("%d %d", &a, &b)
#define sfff(a,b,c) scanf("%d %d %d", &a, &b, &c)
#define pb          push_back

#define hp          (LL) 999983
#define MAX         100000
typedef long long int LL;

const int inf = 1e9;
int rnk[MAX+11], n;
int dp[MAX+11][6];

int calc(int pos, int r)
{
    assert(r > 0 && r <= 4);

    if(rnk[pos] < r) return inf;
    if(100 - rnk[pos] < 4 - r) return inf;
    if(pos == n) return 4-r;

    if(dp[pos][r] != -1) return dp[pos][r];

    int ret = 4-r + calc(pos+1, 1), nr, adp, df;

    for(nr = 2; nr <= 4; nr++)
    {
        if(nr <= r) ret = min(ret, 4-r + nr-1 + calc(pos+1, nr));
        else
        {
            adp = nr - r - 1;
            if(rnk[pos+1] - rnk[pos] - 1 >= adp && rnk[pos] + (adp+1) *10 >= rnk[pos+1]) ret = min(ret, adp + calc(pos+1, nr));
            else ret = min(ret, 4-r + nr-1 + calc(pos+1, nr));
        }
    }

    return dp[pos][r] = ret;
}

int main()
{
    freopen("c:\\Users\\User\\Desktop\\in.txt", "r", stdin);
    freopen("c:\\Users\\User\\Desktop\\out.txt", "w", stdout);

    int i, j, k, t, cs, res;
    int nn, rr;

    sf(t);
    for(cs = 1; cs <= t; cs++)
    {
        memset(dp, -1, sizeof(dp));

        res = inf;

        sf(n);
        for(i = 1; i <= n; i++)
            sf(rnk[i]);


        for(nn = n; nn >= 1; nn--)
            for(rr = 1; rr <= 4; rr++)
                calc(nn, rr);

        res = calc(1,1);
        res = min(res, 1 + calc(1, 2));
        res = min(res, 2 + calc(1, 3));
        res = min(res, 3 + calc(1, 4));

        printf("Case #%d: %d\n", cs, res);
    }
    return 0;
}
